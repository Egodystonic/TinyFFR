material {
	name : %NAME%,
	parameters : [
		{
			type : sampler2d,
			name : color_map
		}
	],
	requires : [
		uv0
	],
	shadingModel : lit,
	flipUV : false,
	postLightingBlending : opaque,
#option AlphaMode
#variant AlphaOff
	blending : opaque
#variant AlphaOn
	blending : fade
#endoption
}

fragment {
	void material(inout MaterialInputs material) {
		vec2 uv0 = getUV0();

		prepareMaterial(material);

#option AlphaMode
#variant AlphaOff
		material.postLightingColor.rgb = texture(materialParams_color_map, uv0).rgb;
#variant AlphaOn
		material.postLightingColor.rgba = texture(materialParams_color_map, uv0).rgba;
#endoption
	}
}
